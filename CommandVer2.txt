แบบ Volume
// ใช้คำสั่งนี้เพื่อแสดงรายการ Volume ที่มีอยู่ในระบบ
docker volume ls 

// ใช้คำสั่งนี้เพื่อสร้าง Volume ใหม่ที่ชื่อว่า wp_volume
docker volume create wp_volume 
docker volume create mysql_volume

// ใช้คำสั่งนี้เพื่อลบ Volume ที่ไม่ต้องการ โดยแทนที่ <volume_name> ด้วยชื่อของ Volume ที่ต้องการลบ
docker volume rm wp_volume mysql_volume
